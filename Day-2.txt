#include <stdio.h>
#define MAXSIZE 10  

int Queue[MAXSIZE];
int front = 0;
int rear = -1;

void insert();
void delete();
void display();

int main() {  
    int choice;

    do {
        printf("\n1. Insert\n2. Delete\n3. Display\n4. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);
        
        switch (choice) {
            case 1:
                insert();
                break;
            case 2:
                delete();
                break;
            case 3:
                display();
                break;
            case 4:
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid choice! Please enter a number between 1 and 4.\n");
        }
    } while (choice != 4);  

    return 0;  
}

void insert() {
    int n;
    if (rear == MAXSIZE - 1) {
        printf("Queue overflow\n");  
    } else {
        printf("Enter the element: ");
        scanf("%d", &n);  
        rear = rear + 1;
        Queue[rear] = n;  
        printf("%d inserted into the queue.\n", n);  
    }
}

void delete() {
    if (front > rear) {  // Queue is empty
        printf("Queue underflow\n");  
    } else {
        printf("%d deleted from the queue.\n", Queue[front]);  // Display the deleted element
        front = front + 1;  // Move front to the next element
    }
}

void display() {
    if (front > rear) {
        printf("Queue is empty\n");
    } else {
        printf("Queue elements are: ");
        for (int i = front; i <= rear; i++) {
            printf("%d ", Queue[i]);
        }
        printf("\n");
    }
}
